  <script>

    const REGION_FILES = {

      "euro-area": "data/euro_area_tvpc_full.csv"

    };


    let fullData = [];


    function loadRegion() {

      let region = document.getElementById("regionSelector").value;

      let file = REGION_FILES[region];


      fetch(file)

        .then(response => response.text())

        .then(text => {

          fullData = parseCSV(text);

          setupControls(fullData);

          document.getElementById("controls").style.display = "block";

          updatePlot();

        });

    }


    function parseCSV(text) {

      let lines = text.split("\n");

      let headers = lines[0].split(",");


      return lines.slice(1).map(line => {

        let cols = line.split(",");

        let obj = {};

        headers.forEach((h, i) => obj[h] = cols[i]);

        return obj;

      }).filter(row => row.date); // drop empty

    }


    function setupControls(data) {

      // Bandwidth options

      let bwSet = [...new Set(data.map(d => d.h1))];

      let bwSel = document.getElementById("bandwidth");

      bwSel.innerHTML = "";

      bwSet.forEach(h => {

        let opt = document.createElement("option");

        opt.value = h;

        opt.textContent = h;

        bwSel.appendChild(opt);

      });


      // Coefficient options

      let coefSet = [...new Set(data.map(d => d.coef))];

      let coefSel = document.getElementById("coef");

      coefSel.innerHTML = "";

      coefSet.forEach(c => {

        let opt = document.createElement("option");

        opt.value = c;

        opt.textContent = c;

        coefSel.appendChild(opt);

      });


      // Add listeners

      document.getElementById("bandwidth").onchange = updatePlot;

      document.getElementById("kernel").onchange = updatePlot;

      document.getElementById("pcType").onchange = updatePlot;

      document.getElementById("estimator").onchange = updatePlot;

      document.getElementById("coef").onchange = updatePlot;

    }


    function updatePlot() {

      let h1      = document.getElementById("bandwidth").value;

      let kernel  = document.getElementById("kernel").value;

      let pcType  = document.getElementById("pcType").value;

      let est     = document.getElementById("estimator").value;

      let coef    = document.getElementById("coef").value;


      let subset = fullData.filter(d =>

         d.h1 == h1 &&

         d.kernel == kernel &&

         d.pc_type == pcType &&

         d.estimator == est &&

         d.coef == coef

      );


      let dates = subset.map(d => d.date);

      let ests  = subset.map(d => parseFloat(d.estimate));

      let lower = subset.map(d => parseFloat(d.lower));

      let upper = subset.map(d => parseFloat(d.upper));


      let trace1 = {

        x: dates,

        y: ests,

        mode: 'lines',

        name: 'Estimate'

      };


      let trace2 = {

        x: [...dates, ...dates.slice().reverse()],

        y: [...upper, ...lower.slice().reverse()],

        fill: 'toself',

        fillcolor: 'rgba(0,150,200,0.2)',

        line: {color: 'transparent'},

        name: 'CI'

      };


      let layout = {

        title: coef + " (" + pcType + ", " + est + ")", 

        showlegend: true

      };


      Plotly.newPlot('plot', [trace2, trace1], layout);

    }
